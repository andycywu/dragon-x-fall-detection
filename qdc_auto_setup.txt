@echo off
setlocal enabledelayedexpansion

echo [INFO] QDC 環境自動設置腳本開始執行...

REM 檢查 Git 是否已安裝
echo [INFO] 檢查 Git 安裝狀態...
where git >nul 2>&1
if %errorlevel% neq 0 (
    echo [WARN] Git 未安裝，正在下載安裝程序...
    REM 下載 Git 安裝程序
    powershell -Command "Invoke-WebRequest -Uri 'https://github.com/git-for-windows/git/releases/download/v2.40.0.windows.1/Git-2.40.0-64-bit.exe' -OutFile '%TEMP%\git-installer.exe'"
    echo [INFO] 正在安裝 Git，請稍候...
    %TEMP%\git-installer.exe /VERYSILENT /NORESTART
    echo [INFO] Git 安裝完成，請重新運行此腳本
    exit /b 1
) else (
    echo [SUCCESS] Git 已安裝: 
    git --version
)

REM 檢查 Python 是否已安裝
echo [INFO] 檢查 Python 安裝狀態...
where python >nul 2>&1
if %errorlevel% neq 0 (
    echo [WARN] Python 未安裝，正在下載安裝程序...
    REM 下載 Python 安裝程序
    powershell -Command "Invoke-WebRequest -Uri 'https://www.python.org/ftp/python/3.10.11/python-3.10.11-amd64.exe' -OutFile '%TEMP%\python-installer.exe'"
    echo [INFO] 正在安裝 Python，請稍候...
    %TEMP%\python-installer.exe /quiet InstallAllUsers=1 PrependPath=1
    echo [INFO] Python 安裝完成，請重新運行此腳本
    exit /b 1
) else (
    echo [SUCCESS] Python 已安裝:
    python --version
)

REM 克隆倉庫
echo [INFO] 正在克隆 GitHub 倉庫...
if not exist "C:\dragon-x-fall-detection" (
    cd C:\
    git clone https://github.com/andycywu/dragon-x-fall-detection.git
    if %errorlevel% neq 0 (
        echo [ERROR] 克隆倉庫失敗
        exit /b 1
    )
    echo [SUCCESS] 倉庫克隆成功
) else (
    echo [INFO] 倉庫已存在，正在更新...
    cd C:\dragon-x-fall-detection
    git pull
    echo [SUCCESS] 倉庫更新完成
)

REM 安裝 Python 套件
echo [INFO] 安裝 Python 套件...
cd C:\dragon-x-fall-detection
pip install numpy opencv-python onnxruntime
pip install -r requirements.txt
pip install -U qai-hub qai-hub-models "protobuf==4.25.3"

REM 設置 QAI Hub 配置
echo [INFO] 設置 QAI Hub 配置...
if not exist "%USERPROFILE%\.qai_hub" mkdir "%USERPROFILE%\.qai_hub"
echo [default] > "%USERPROFILE%\.qai_hub\client.ini"
echo api_token = pcu8nz63e4j3nzqgy7tjzvr2dmpc01cocltahr0d >> "%USERPROFILE%\.qai_hub\client.ini"
echo api_key = pcu8nz63e4j3nzqgy7tjzvr2dmpc01cocltahr0d >> "%USERPROFILE%\.qai_hub\client.ini"
echo base_api_url = https://app.aihub.qualcomm.com >> "%USERPROFILE%\.qai_hub\client.ini"
echo web_url = https://app.aihub.qualcomm.com >> "%USERPROFILE%\.qai_hub\client.ini"

REM 測試 QAI Hub 連接
echo [INFO] 測試 QAI Hub 連接...
python -c "import qai_hub as hub; print('可用設備:', [d.name for d in hub.get_devices()])" 2>nul
if %errorlevel% neq 0 (
    echo [WARN] QAI Hub 連接失敗，請手動檢查配置
) else (
    echo [SUCCESS] QAI Hub 連接成功
)

echo.
echo [SUCCESS] 環境設置完成！
echo.
echo [INFO] 現在您可以運行:
echo     cd C:\dragon-x-fall-detection
echo     python dragon_x_fall_detection_system.py
echo.

endlocal
